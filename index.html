const countCharacters = (S) => 
        {
           let countA=0;
            let countD=0;
            for(let i=0;i<=S.length;i++){
              if(S.charAt(i)==='A'){
                countA++;
              }
              else if(S.charAt(i)==='D'){
                countD++;
              }
            }
            let ans = [countA,countD];
            return ans;  
        };

var Count_Occ = (s) => 
        {
          let n=s.length;
          let str=s.split("").sort().join("");
          let ans="";
          for(let i=0;i< n;i++){
            let count=0;
            for(let j=0;j< n;j++){
              if(str[i]===str[j] && i>j){
                break;
              }
              if(str[i]===str[j]){
                count+=1;
              }
            }
             if(count>1){
            ans+=str[i]+count;
            }
          }
          return ans;
        };

var Palin_Check = (str) =>
        {
         let ans="";
          for(let i=str.length-1;i>=0;i--){
            ans+=str[i];
          }
          if(ans===str){
            return "True"; 
          }
          else{
             return "False";
          }	 
        };
